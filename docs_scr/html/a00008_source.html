49 public static void GetAllUsers()
50 {
51 string users = Processes.Execute("wmic", "useraccount list full");
52 string temp_file = $@"{Path.GetTempPath()}\Пользователи.txt";
53 FileSystem.WriteTextToFileUTF8(users, temp_file);
54 string[] lines = File.ReadAllLines(temp_file);
55 using (StreamWriter writer = new StreamWriter(temp_file))
56 {
57 foreach (string line in lines)
58 {
59 if (!string.IsNullOrWhiteSpace(line))
60 writer.WriteLine(line);
61 }
62 }
63 string[] rep = File.ReadAllLines(temp_file);
64 for (int i = 0; i < rep.Length; i++)
65 {
66 if (i == 0)
67 {
68 rep[0] = "Пользователи:\n\n" + rep[0];
69 }
70 rep[i] = rep[i].Replace("TRUE", "Да");
71 rep[i] = rep[i].Replace("FALSE", "Нет");
72 rep[i] = rep[i].Replace("Description", "Описание");
73 rep[i] = rep[i].Replace("Domain", "Домен");
74 rep[i] = rep[i].Replace("FullName", "Полное имя");
75 rep[i] = rep[i].Replace("Name", "Имя");
76 rep[i] = rep[i].Replace("LocalAccount", "Локальная запись");
77 rep[i] = rep[i].Replace("Disabled", "Отключена");
78 rep[i] = rep[i].Replace("Status", "Статус");
79 rep[i] = rep[i].Replace("Lockout", "Блокировка");

156 public static string GetSize()
157 {
158 string size = string.Empty;
159 Graphics graphics = Graphics.FromHwnd(IntPtr.Zero);
160 IntPtr desktop = graphics.GetHdc();
161 int monitorHeight = Win32.Import.GetDeviceCaps(desktop, 6);
162 int monitorWidth = Win32.Import.GetDeviceCaps(desktop, 4);
163 size = $"{Math.Sqrt(Math.Pow(monitorHeight, 2) + Math.Pow(monitorWidth, 2)) / 25,4:#,##0.00}";
164 return size;
165 }
