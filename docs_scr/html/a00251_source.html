43 public bool PreFilterMessage(ref Message m)
44 {
45 // Exit out early if we're not dragging any content
46 if (!_isDragging)
47 return false;
48
49 // We only care about mouse events
50 if (!(m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.MOUSEMOVE ||
51 m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.LBUTTONDOWN || m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.LBUTTONUP || m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.LBUTTONDBLCLK ||
52 m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.RBUTTONDOWN || m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.RBUTTONUP || m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.RBUTTONDBLCLK))
53 return false;
54
55 // Move content
56 if (m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.MOUSEMOVE)
57 {
58 HandleDrag();
59 return false;
60 }
61
62 // Drop content
63 if (m.Msg == (int)Functions.Windows.Win32.Enums.WINDOW_MESSAGE.LBUTTONUP)
64 {
65 if (_targetRegion != null)
66 {
67 _dockPanel.RemoveContent(_dragContent);
68 _dragContent.DockArea = _targetRegion.DockArea;
69 _dockPanel.AddContent(_dragContent);
70 }
71 else if (_targetGroup != null)
72 {
73 _dockPanel.RemoveContent(_dragContent);
74
75 switch (_insertType)
76 {
77 case DockInsertType.None:
78 _dockPanel.AddContent(_dragContent, _targetGroup);
79 break;
80
81 case DockInsertType.Before:
82 case DockInsertType.After:
83 _dockPanel.InsertContent(_dragContent, _targetGroup, _insertType);
84 break;
85 }
86 }
87
88 StopDrag();
89 return false;
90 }
91
92 return true;
93 }

99 public void StartDrag(VitNX_DockContent content)
100 {
101 _regionDropAreas = new Dictionary<VitNX_DockRegion, DockDropArea>();
102 _groupDropAreas = new Dictionary<VitNX_DockGroup, DockDropCollection>();
103
104 // Add all regions and groups to the drop collections
105 foreach (var region in _dockPanel.Regions.Values)
106 {
107 if (region.DockArea == VitNX_DockArea.Document)
108 continue;
109
110 // If the region is visible then build drop areas for the groups.
111 if (region.Visible)
112 {
113 foreach (var group in region.Groups)
114 {
115 var collection = new DockDropCollection(_dockPanel, group);
116 _groupDropAreas.Add(group, collection);
117 }
118 }
119 // If the region is NOT visible then build the drop area for the region itself.
120 else
121 {
122 var area = new DockDropArea(_dockPanel, region);
123 _regionDropAreas.Add(region, area);
124 }
125 }
126
127 _dragContent = content;
128 _isDragging = true;
129 }

158 private void HandleDrag()
159 {
160 var location = Cursor.Position;
161
162 _insertType = DockInsertType.None;
163
164 _targetRegion = null;
165 _targetGroup = null;
166
167 // Check all region drop areas
168 foreach (var area in _regionDropAreas.Values)
169 {
170 if (area.DropArea.Contains(location))
171 {
172 _insertType = DockInsertType.None;
173 _targetRegion = area.DockRegion;
174 UpdateHighlightForm(area.HighlightArea);
175 return;
176 }
177 }
178
179 // Check all group drop areas
180 foreach (var collection in _groupDropAreas.Values)
181 {
182 var sameRegion = false;
183 var sameGroup = false;
184 var groupHasOtherContent = false;
185
186 if (collection.DropArea.DockGroup == _dragContent.DockGroup)
187 sameGroup = true;
188
189 if (collection.DropArea.DockGroup.DockRegion == _dragContent.DockRegion)
190 sameRegion = true;
191
192 if (_dragContent.DockGroup.ContentCount > 1)
193 groupHasOtherContent = true;
194
195 // If we're hovering over the group itself, only allow inserting before/after if multiple content is tabbed.
196 if (!sameGroup || groupHasOtherContent)
197 {
198 var skipBefore = false;
199 var skipAfter = false;
200
201 // Inserting before/after other content might cause the content to be dropped on to its own location.
202 // Check if the group above/below the hovered group contains our drag content.
203 if (sameRegion && !groupHasOtherContent)
204 {
205 if (collection.InsertBeforeArea.DockGroup.Order == _dragContent.DockGroup.Order + 1)
206 skipBefore = true;
207
208 if (collection.InsertAfterArea.DockGroup.Order == _dragContent.DockGroup.Order - 1)
209 skipAfter = true;
210 }
211
212 if (!skipBefore)
213 {
214 if (collection.InsertBeforeArea.DropArea.Contains(location))
215 {
216 _insertType = DockInsertType.Before;
217 _targetGroup = collection.InsertBeforeArea.DockGroup;
218 UpdateHighlightForm(collection.InsertBeforeArea.HighlightArea);
219 return;
220 }
221 }
222
223 if (!skipAfter)
224 {
225 if (collection.InsertAfterArea.DropArea.Contains(location))
226 {
227 _insertType = DockInsertType.After;
228 _targetGroup = collection.InsertAfterArea.DockGroup;
229 UpdateHighlightForm(collection.InsertAfterArea.HighlightArea);
230 return;
231 }
232 }
233 }
234
235 // Don't allow content to be dragged on to itself
236 if (!sameGroup)
237 {
238 if (collection.DropArea.DropArea.Contains(location))
239 {
240 _insertType = DockInsertType.None;
241 _targetGroup = collection.DropArea.DockGroup;
242 UpdateHighlightForm(collection.DropArea.HighlightArea);
243 return;
244 }
245 }
246 }
247
248 // Not hovering over anything - hide the highlight
249 if (_highlightForm.Visible)
250 _highlightForm.Hide();
251
252 // Show we can't drag here
253 Cursor.Current = Cursors.No;
254 }
