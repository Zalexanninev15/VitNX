45 public static string WriteAllUsersToTempFile()
46 {
47 string users = Processes.Execute("wmic", "useraccount list full");
48 string tempFile = $@"{Path.GetTempPath()}\Users.txt";
49 FileSystem.WriteTextToFileUTF8(users, tempFile);
50 string[] lines = File.ReadAllLines(tempFile);
51 using (StreamWriter writer = new StreamWriter(tempFile))
52 {
53 foreach (string line in lines)
54 {
55 if (!string.IsNullOrWhiteSpace(line))
56 writer.WriteLine(line);
57 }
58 }
59 string[] usersList = File.ReadAllLines(tempFile);
60 for (int i = 0; i < usersList.Length; i++)
61 {
62 if (i == 0)
63 usersList[0] = "Users:\n\n" + usersList[0];
64 usersList[i] = usersList[i].Replace("TRUE", "Yes");
65 usersList[i] = usersList[i].Replace("FALSE", "No");
66 usersList[i] = usersList[i].Replace("FullName", "Full name");
67 usersList[i] = usersList[i].Replace("LocalAccount", "Local account");
68 usersList[i] = usersList[i].Replace("Degraded", "Degraded\n");
69 usersList[i] = usersList[i].Replace("InstallDate", "Creation date");
70 usersList[i] = usersList[i].Replace("AccountType", "Account type");
71 usersList[i] = usersList[i].Replace("PasswordChangeable", "Password changeable");
72 usersList[i] = usersList[i].Replace("PasswordExpires", "Password expires");
73 usersList[i] = usersList[i].Replace("PasswordRequired", "Password required");
74 usersList[i] = usersList[i].Replace("512", "Plain [512]").Replace("2", "Disconnected [2]").Replace("256", "Temporary duplicate account [256]").Replace("128", "Encrypted password allowed [128]");
75 usersList[i] = usersList[i].Replace("OK", "Enabled\n");
76 }
77 File.WriteAllLines(tempFile, usersList);
78 return tempFile;

200 private static string SizeSuffix(Int64 value)
201 {
202 if (value < 0)
203 return "-" + SizeSuffix(-value);
204 if (value == 0)
205 return "0.0 bytes";
206 int mag = (int)Math.Log(value, 1024);
207 decimal adjustedSize = (decimal)value / (1L << (mag * 10));
208 return string.Format("{0:n1} {1}", adjustedSize, Helper.SizeSuffixes[mag]);
209 }
