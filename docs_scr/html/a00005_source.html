209 private static string SizeSuffix(Int64 value)
210 {
211 if (value < 0)
212 return "-" + SizeSuffix(-value);
213 if (value == 0)
214 return "0.0 bytes";
215 int mag = (int)Math.Log(value, 1024);
216 decimal adjustedSize = (decimal)value / (1L << (mag * 10));
217 return string.Format("{0:n1} {1}", adjustedSize, Helper._sizeSuffixes[mag]);
218 }
