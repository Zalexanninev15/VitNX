72 private GraphicsPath GetFigurePath(Rectangle rect, int radius)
73 {
74 GraphicsPath path = new GraphicsPath();
75 float curveSize = radius * 2F; path.StartFigure();
76 path.AddArc(rect.X, rect.Y, curveSize, curveSize, 180, 90);
77 path.AddArc(rect.Right - curveSize, rect.Y, curveSize, curveSize, 270, 90);
78 path.AddArc(rect.Right - curveSize, rect.Bottom - curveSize, curveSize, curveSize, 0, 90);
79 path.AddArc(rect.X, rect.Bottom - curveSize, curveSize, curveSize, 90, 90);
80 path.CloseFigure();
