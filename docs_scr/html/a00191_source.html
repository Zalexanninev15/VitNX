67 {

69 dockContent.Dock = DockStyle.Fill;

71 dockContent.Order = 0;
72
73 if (_contents.Count > 0)

76 foreach (var otherContent in _contents)
77 {
78 if (otherContent.Order >= order)
79 order = otherContent.Order + 1;

82 dockContent.Order = order;

86 Controls.Add(dockContent);

99 dockContent.Visible = false;

103 menuItem.Tag = dockContent;
104 menuItem.Click += TabMenuItem_Select;
105 menuItem.Image = dockContent.Icon;

156 private void UpdateTabArea()
157 {
158 if (DockArea == VitNX_DockArea.Document)
159 _tabArea.Visible = (_contents.Count > 0);
160 else
161 _tabArea.Visible = (_contents.Count > 1);
162
163 var size = 0;
164
165 switch (DockArea)
166 {
167 case VitNX_DockArea.Document:
168 size = _tabArea.Visible ? Constsants.DocumentTabAreaSize : 0;
169 Padding = new Padding(0, size, 0, 0);
170 _tabArea.ClientRectangle = new Rectangle(Padding.Left, 0, ClientRectangle.Width - Padding.Horizontal, size);
171 break;
172
173 case VitNX_DockArea.Left:
174 case VitNX_DockArea.Right:
175 size = _tabArea.Visible ? Constsants.ToolWindowTabAreaSize : 0;
176 Padding = new Padding(0, 0, 0, size);
177 _tabArea.ClientRectangle = new Rectangle(Padding.Left, ClientRectangle.Bottom - size, ClientRectangle.Width - Padding.Horizontal, size);
178 break;
179
180 case VitNX_DockArea.Bottom:
181 size = _tabArea.Visible ? Constsants.ToolWindowTabAreaSize : 0;
182 Padding = new Padding(1, 0, 0, size);
183 _tabArea.ClientRectangle = new Rectangle(Padding.Left, ClientRectangle.Bottom - size, ClientRectangle.Width - Padding.Horizontal, size);
184 break;
185 }
186
187 if (DockArea == VitNX_DockArea.Document)
188 {
189 var dropdownSize = Constsants.DocumentTabAreaSize;
190 _tabArea.DropdownRectangle = new Rectangle(_tabArea.ClientRectangle.Right - dropdownSize, 0, dropdownSize, dropdownSize);
191 }
192
193 BuildTabs();
194
195 EnsureVisible();

198 private void BuildTabs()
199 {
200 if (!_tabArea.Visible)
201 return;
202
203 SuspendLayout();
204
205 var closeButtonSize = DockIcons.close.Width;
206
207 // Calculate areas of all tabs
208 var totalSize = 0;
209
210 var orderedContent = _contents.OrderBy(c => c.Order);
211
212 foreach (var content in orderedContent)
213 {
214 int width;
215
216 var tab = _tabs[content];
217
218 using (var g = CreateGraphics())
219 {
220 width = tab.CalculateWidth(g, Font);
221 }
222
223 // Add additional width for document tab items
224 if (DockArea == VitNX_DockArea.Document)
225 {
226 width += 5;
227 width += closeButtonSize;
228
229 if (tab.DockContent.Icon != null)
230 width += tab.DockContent.Icon.Width + 5;
231 }
232
233 // Show separator on all tabs for now
234 tab.ShowSeparator = true;
235 width += 1;
236
237 var y = DockArea == VitNX_DockArea.Document ? 0 : ClientRectangle.Height - Constsants.ToolWindowTabAreaSize;
238 var height = DockArea == VitNX_DockArea.Document ? Constsants.DocumentTabAreaSize : Constsants.ToolWindowTabAreaSize;
239
240 var tabRect = new Rectangle(_tabArea.ClientRectangle.Left + totalSize, y, width, height);
241 tab.ClientRectangle = tabRect;
242
243 totalSize += width;
244 }
245
246 // Cap the size if too large for the tab area
247 if (DockArea != VitNX_DockArea.Document)
248 {
249 if (totalSize > _tabArea.ClientRectangle.Width)
250 {
251 var difference = totalSize - _tabArea.ClientRectangle.Width;
252
253 // No matter what, we want to slice off the 1 pixel separator from the final tab.
254 var lastTab = _tabs[orderedContent.Last()];
255 var tabRect = lastTab.ClientRectangle;
256 lastTab.ClientRectangle = new Rectangle(tabRect.Left, tabRect.Top, tabRect.Width - 1, tabRect.Height);
257 lastTab.ShowSeparator = false;
258
259 var differenceMadeUp = 1;
260
261 // Loop through and progressively resize the larger tabs until the total size fits within the tab area.
262 while (differenceMadeUp < difference)
263 {
264 var largest = _tabs.Values.OrderByDescending(tab => tab.ClientRectangle.Width)
265 .First()
266 .ClientRectangle.Width;
267
268 foreach (var content in orderedContent)
269 {
270 var tab = _tabs[content];
271
272 // Check if previous iteration of loop met the difference
273 if (differenceMadeUp >= difference)
274 break;
275
276 if (tab.ClientRectangle.Width >= largest)
277 {
278 var rect = tab.ClientRectangle;
279 tab.ClientRectangle = new Rectangle(rect.Left, rect.Top, rect.Width - 1, rect.Height);
280 differenceMadeUp += 1;
281 }
282 }
283 }
284
285 // After resizing the tabs reposition them accordingly.
286 var xOffset = 0;
287 foreach (var content in orderedContent)
288 {
289 var tab = _tabs[content];
290
291 var rect = tab.ClientRectangle;
292 tab.ClientRectangle = new Rectangle(_tabArea.ClientRectangle.Left + xOffset, rect.Top, rect.Width, rect.Height);
293
294 xOffset += rect.Width;
295 }
296 }
297 }
298
299 // Build close button rectangles
300 if (DockArea == VitNX_DockArea.Document)
301 {
302 foreach (var content in orderedContent)
303 {
304 var tab = _tabs[content];
305 var closeRect = new Rectangle(tab.ClientRectangle.Right - 7 - closeButtonSize - 1,
306 tab.ClientRectangle.Top + (tab.ClientRectangle.Height / 2) - (closeButtonSize / 2) - 1,
307 closeButtonSize, closeButtonSize);
308 tab.CloseButtonRectangle = closeRect;
309 }
310 }
311
312 // Update the tab area with the new total tab width
313 totalSize = 0;
314 foreach (var content in orderedContent)
315 {
316 var tab = _tabs[content];
317 totalSize += tab.ClientRectangle.Width;
318 }
319
320 _tabArea.TotalTabSize = totalSize;
321
322 ResumeLayout();
323
324 Invalidate();
325 }

327 public void EnsureVisible()
328 {
329 if (DockArea != VitNX_DockArea.Document)
330 return;
331
332 if (VisibleContent == null)
333 return;
334
335 var width = ClientRectangle.Width - Padding.Horizontal - _tabArea.DropdownRectangle.Width;
336 var offsetArea = new Rectangle(Padding.Left, 0, width, 0);
337
338 var tab = _tabs[VisibleContent];
339
340 if (tab.ClientRectangle.IsEmpty)
341 return;
342
343 if (RectangleToTabArea(tab.ClientRectangle).Left < offsetArea.Left)
344 _tabArea.Offset = tab.ClientRectangle.Left;
345 else if (RectangleToTabArea(tab.ClientRectangle).Right > offsetArea.Right)
346 _tabArea.Offset = tab.ClientRectangle.Right - width;
347
348 if (_tabArea.TotalTabSize < offsetArea.Width)
349 _tabArea.Offset = 0;
350
351 if (_tabArea.TotalTabSize > offsetArea.Width)
352 {
353 var orderedContent = _contents.OrderBy(x => x.Order);
354 var lastTab = _tabs[orderedContent.Last()];
355 if (lastTab != null)
356 {
357 if (RectangleToTabArea(lastTab.ClientRectangle).Right < offsetArea.Right)
358 _tabArea.Offset = lastTab.ClientRectangle.Right - width;
359 }
360 }
361
362 Invalidate();
363 }

406 protected override void OnMouseMove(MouseEventArgs e)
407 {
408 base.OnMouseMove(e);
409
410 if (_dragTab != null)
411 {
412 var offsetX = e.Location.X + _tabArea.Offset;
413 if (offsetX < _dragTab.ClientRectangle.Left)
414 {
415 if (_dragTab.DockContent.Order > 0)
416 {
417 var otherTabs = _tabs.Values.Where(t => t.DockContent.Order == _dragTab.DockContent.Order - 1).ToList();
418 if (otherTabs.Count == 0)
419 return;
420
421 var otherTab = otherTabs.First();
422
423 if (otherTab == null)
424 return;
425
426 var oldIndex = _dragTab.DockContent.Order;
427 _dragTab.DockContent.Order = oldIndex - 1;
428 otherTab.DockContent.Order = oldIndex;
429
430 BuildTabs();
431 EnsureVisible();
432
433 _tabArea.RebuildMenu();
434
435 return;
436 }
437 }
438 else if (offsetX > _dragTab.ClientRectangle.Right)
439 {
440 var maxOrder = _contents.Count;
441
442 if (_dragTab.DockContent.Order < maxOrder)
443 {
444 var otherTabs = _tabs.Values.Where(t => t.DockContent.Order == _dragTab.DockContent.Order + 1).ToList();
445 if (otherTabs.Count == 0)
446 return;
447
448 var otherTab = otherTabs.First();
449
450 if (otherTab == null)
451 return;
452
453 var oldIndex = _dragTab.DockContent.Order;
454 _dragTab.DockContent.Order = oldIndex + 1;
455 otherTab.DockContent.Order = oldIndex;
456
457 BuildTabs();
458 EnsureVisible();
459
460 _tabArea.RebuildMenu();
461
462 return;
463 }
464 }
465
466 return;
467 }
468
469 if (_tabArea.DropdownRectangle.Contains(e.Location))
470 {
471 _tabArea.DropdownHot = true;
472
473 foreach (var tab in _tabs.Values)
474 tab.Hot = false;
475
476 Invalidate();
477 return;
478 }
479
480 _tabArea.DropdownHot = false;
481
482 foreach (var tab in _tabs.Values)
483 {
484 var rect = RectangleToTabArea(tab.ClientRectangle);
485 var hot = rect.Contains(e.Location);
486
487 if (tab.Hot != hot)
488 {
489 tab.Hot = hot;
490 Invalidate();
491 }
492
493 var closeRect = RectangleToTabArea(tab.CloseButtonRectangle);
494 var closeHot = closeRect.Contains(e.Location);
495
496 if (tab.CloseButtonHot != closeHot)
497 {
498 tab.CloseButtonHot = closeHot;
499 Invalidate();
500 }
501 }

683 {
684 var tabRect = RectangleToTabArea(tab.ClientRectangle);
685
686 var isVisibleTab = VisibleContent == tab.DockContent;
687 var isActiveGroup = DockPanel.ActiveGroup == this;
688
689 var bgColor = isVisibleTab ? Colors.BlueSelection : Colors.VitNXBackground;
690
691 if (!isActiveGroup)
692 bgColor = isVisibleTab ? Colors.GreySelection : Colors.VitNXBackground;
693
694 if (tab.Hot && !isVisibleTab)
695 bgColor = Colors.MediumBackground;
696
697 using (var b = new SolidBrush(bgColor))

699 g.FillRectangle(b, tabRect);
700 }
701
702 // Draw separators
703 if (tab.ShowSeparator)
704 {
705 using (var p = new Pen(Colors.VitNXBorder))
706 {
707 g.DrawLine(p, tabRect.Right - 1, tabRect.Top, tabRect.Right - 1, tabRect.Bottom);
708 }
709 }
710
711 var xOffset = 0;
712
713 // Draw icon
714 if (tab.DockContent.Icon != null)
715 {
716 g.DrawImageUnscaled(tab.DockContent.Icon, tabRect.Left + 5, tabRect.Top + 4);
717 xOffset += tab.DockContent.Icon.Width + 2;
718 }
719
720 var tabTextFormat = new StringFormat
721 {
722 Alignment = StringAlignment.Near,
723 LineAlignment = StringAlignment.Center,
724 FormatFlags = StringFormatFlags.NoWrap,
725 Trimming = StringTrimming.EllipsisCharacter
726 };
727
728 // Draw text
729 var textColor = isVisibleTab ? Colors.LightText : Colors.DisabledText;
730 using (var b = new SolidBrush(textColor))
731 {
732 var textRect = new Rectangle(tabRect.Left + 5 + xOffset, tabRect.Top, tabRect.Width - tab.CloseButtonRectangle.Width - 7 - 5 - xOffset, tabRect.Height);
733 g.DrawString(tab.DockContent.DockText, Font, b, textRect, tabTextFormat);
734 }
735
736 // Close button
737 var img = tab.CloseButtonHot ? DockIcons.inactive_close_selected : DockIcons.inactive_close;
738
739 if (isVisibleTab)

741 if (isActiveGroup)

748 g.DrawImageUnscaled(img, closeRect.Left, closeRect.Top);
