104 private void SetButtons()
105 {
106 foreach (var btn in _buttons)
107 btn.Visible = false;
108
109 switch (_dialogButtons)

111 case VitNX_DialogButton.Ok:
112 ShowButton(btnOk, true);
113 AcceptButton = btnOk;

116 case VitNX_DialogButton.Close:
117 ShowButton(btnClose, true);
118 AcceptButton = btnClose;
119 CancelButton = btnClose;
120 break;
121
122 case VitNX_DialogButton.OkCancel:
123 ShowButton(btnOk);
124 ShowButton(btnCancel, true);
125 AcceptButton = btnOk;
126 CancelButton = btnCancel;
127 break;
128
129 case VitNX_DialogButton.AbortRetryIgnore:
130 ShowButton(btnAbort);
131 ShowButton(btnRetry);
132 ShowButton(btnIgnore, true);
133 AcceptButton = btnAbort;
134 CancelButton = btnIgnore;
135 break;
136
137 case VitNX_DialogButton.RetryCancel:
138 ShowButton(btnRetry);
139 ShowButton(btnCancel, true);
140 AcceptButton = btnRetry;
141 CancelButton = btnCancel;
142 break;
143
144 case VitNX_DialogButton.YesNo:
145 ShowButton(btnYes);
146 ShowButton(btnNo, true);
147 AcceptButton = btnYes;
148 CancelButton = btnNo;
149 break;
150
151 case VitNX_DialogButton.YesNoCancel:
152 ShowButton(btnYes);
153 ShowButton(btnNo);
154 ShowButton(btnCancel, true);
155 AcceptButton = btnYes;
