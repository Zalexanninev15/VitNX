16 {
17 string procList = "All processes:";
18 foreach (Process process in Process.GetProcesses())
19 {
20 var procSize = "0x0";
21 try
22 {
23 var counter = new PerformanceCounter("Process", "Working Set - Private", process.ProcessName);
24 procSize = Convert.ToString(counter.RawValue / 1048576);
25 }
26 catch { }
27 if (procSize != "0x0" && procSize != "0")
28 procList += $"\n\nName: {process.ProcessName}.exe" +
29 $"\nID: {process.Id}" +
30 $"\nUsed memory: {procSize} МБ" +
31 $"\nTitle & Handle: \"{process.MainWindowTitle}\" & {process.MainWindowHandle}";
32 }
33 return procList;

59 public static string Execute(string targetFile,
60 string arguments)
61 {
62 var codepage = System.Globalization.CultureInfo.CurrentCulture.TextInfo.OEMCodePage;
63 var desiredEncoding = System.Text.Encoding.GetEncoding(codepage);
64 var start = new Process
65 {
66 StartInfo = new ProcessStartInfo
67 {
68 FileName = targetFile,
69 Arguments = arguments,
70 CreateNoWindow = true,
71 UseShellExecute = false,
72 //Verb = "open",
73 WindowStyle = ProcessWindowStyle.Hidden,
74 RedirectStandardOutput = true,
75 StandardOutputEncoding = desiredEncoding
76 }
77 };
78 start.WaitForExit();
79 start.Start();
80 string output = start.StandardOutput.ReadToEnd();
81 return output;

171 public static string GetPath(string applicationName)
172 {
173 var installPath = FindAppPath(@"SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall", applicationName);
174 if (installPath == "")
175 installPath = FindAppPath(@"SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall", applicationName);
176 if (installPath == "")
177 installPath = FindAppPath(@"HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Uninstall", applicationName);
178 return installPath;
179 }
180
181 public static string FindAppPath(string keyPath, string applicationName)
182 {
183 var uninstall = Registry.LocalMachine.OpenSubKey(keyPath);
184 foreach (var productSubKey in uninstall.GetSubKeyNames())
185 {
186 var product = uninstall.OpenSubKey(productSubKey);
187 var displayName = product.GetValue("DisplayName");
188 if (Convert.ToString(displayName) != "" && Convert.ToString(displayName).Contains(applicationName))
189 return product.GetValue("InstallLocation").ToString();
190 }
191 return null;
192 }
193 }
