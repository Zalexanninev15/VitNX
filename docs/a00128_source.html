172 protected override void OnMouseDown(MouseEventArgs e)
173 {
174 base.OnMouseDown(e);
175
176 if (_thumbArea.Contains(e.Location) && e.Button == MouseButtons.Left)
177 {
178 _isScrolling = true;
179 _initialContact = e.Location;
180
181 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
182 _initialValue = _thumbArea.Top;
183 else
184 _initialValue = _thumbArea.Left;
185
186 Invalidate();
187 return;
188 }
189
190 if (_upArrowArea.Contains(e.Location) && e.Button == MouseButtons.Left)
191 {
192 _upArrowClicked = true;
193 _scrollTimer.Enabled = true;
194
195 Invalidate();
196 return;
197 }
198
199 if (_downArrowArea.Contains(e.Location) && e.Button == MouseButtons.Left)
200 {
201 _downArrowClicked = true;
202 _scrollTimer.Enabled = true;
203
204 Invalidate();
205 return;
206 }
207
208 if (_trackArea.Contains(e.Location) && e.Button == MouseButtons.Left)
209 {
210 // Step 1. Check if our input is at least aligned with the thumb
211 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
212 {
213 var modRect = new Rectangle(_thumbArea.Left, _trackArea.Top, _thumbArea.Width, _trackArea.Height);
214 if (!modRect.Contains(e.Location))
215 return;
216 }
217 else if (_scrollOrientation == VitNX_ScrollOrientation.Horizontal)
218 {
219 var modRect = new Rectangle(_trackArea.Left, _thumbArea.Top, _trackArea.Width, _thumbArea.Height);
220 if (!modRect.Contains(e.Location))
221 return;
222 }
223
224 // Step 2. Scroll to the area initially clicked.
225 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
226 {
227 var loc = e.Location.Y;
228 loc -= _upArrowArea.Bottom - 1;
229 loc -= _thumbArea.Height / 2;
230 ScrollToPhysical(loc);
231 }
232 else
233 {
234 var loc = e.Location.X;
235 loc -= _upArrowArea.Right - 1;
236 loc -= _thumbArea.Width / 2;
237 ScrollToPhysical(loc);
238 }
239
240 // Step 3. Initiate a thumb drag.
241 _isScrolling = true;
242 _initialContact = e.Location;
243 _thumbHot = true;
244
245 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
246 _initialValue = _thumbArea.Top;
247 else
248 _initialValue = _thumbArea.Left;
249
250 Invalidate();
251 return;
252 }

268 protected override void OnMouseMove(MouseEventArgs e)
269 {
270 base.OnMouseMove(e);
271
272 if (!_isScrolling)
273 {
274 var thumbHot = _thumbArea.Contains(e.Location);
275 if (_thumbHot != thumbHot)
276 {
277 _thumbHot = thumbHot;
278 Invalidate();
279 }
280
281 var upArrowHot = _upArrowArea.Contains(e.Location);
282 if (_upArrowHot != upArrowHot)
283 {
284 _upArrowHot = upArrowHot;
285 Invalidate();
286 }
287
288 var downArrowHot = _downArrowArea.Contains(e.Location);
289 if (_downArrowHot != downArrowHot)
290 {
291 _downArrowHot = downArrowHot;
292 Invalidate();
293 }
294 }
295
296 if (_isScrolling)
297 {
298 if (e.Button != MouseButtons.Left)
299 {
300 OnMouseUp(null);
301 return;
302 }
303
304 var difference = new Point(e.Location.X - _initialContact.X, e.Location.Y - _initialContact.Y);
305
306 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
307 {
308 var thumbPos = (_initialValue - _trackArea.Top);
309 var newPosition = thumbPos + difference.Y;
310
311 ScrollToPhysical(newPosition);
312 }
313 else if (_scrollOrientation == VitNX_ScrollOrientation.Horizontal)
314 {
315 var thumbPos = (_initialValue - _trackArea.Left);
316 var newPosition = thumbPos + difference.X;
317
318 ScrollToPhysical(newPosition);
319 }
320
321 UpdateScrollBar();
322 }
323 }

359 public void ScrollToPhysical(int positionInPixels)
360 {
361 var isVert = _scrollOrientation == VitNX_ScrollOrientation.Vertical;
362
363 var trackAreaSize = isVert ? _trackArea.Height - _thumbArea.Height : _trackArea.Width - _thumbArea.Width;
364
365 var positionRatio = (float)positionInPixels / (float)trackAreaSize;
366 var viewScrollSize = (Maximum - ViewSize);
367
368 var newValue = (int)(positionRatio * viewScrollSize);
369 Value = newValue;
370 }

372 public void ScrollBy(int offset)
373 {

376 }

383
384 var newPosition = thumbPos - offsetInPixels;
385

389 public void UpdateScrollBar()
390 {
391 var area = ClientRectangle;
392
393 // Arrow buttons
394 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
395 {
396 _upArrowArea = new Rectangle(area.Left, area.Top, Constsants.ArrowButtonSize, Constsants.ArrowButtonSize);
397 _downArrowArea = new Rectangle(area.Left, area.Bottom - Constsants.ArrowButtonSize, Constsants.ArrowButtonSize, Constsants.ArrowButtonSize);
398 }
399 else if (_scrollOrientation == VitNX_ScrollOrientation.Horizontal)
400 {
401 _upArrowArea = new Rectangle(area.Left, area.Top, Constsants.ArrowButtonSize, Constsants.ArrowButtonSize);
402 _downArrowArea = new Rectangle(area.Right - Constsants.ArrowButtonSize, area.Top, Constsants.ArrowButtonSize, Constsants.ArrowButtonSize);
403 }
404
405 // Track
406 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
407 {
408 _trackArea = new Rectangle(area.Left, area.Top + Constsants.ArrowButtonSize, area.Width, area.Height - (Constsants.ArrowButtonSize * 2));
409 }
410 else if (_scrollOrientation == VitNX_ScrollOrientation.Horizontal)
411 {
412 _trackArea = new Rectangle(area.Left + Constsants.ArrowButtonSize, area.Top, area.Width - (Constsants.ArrowButtonSize * 2), area.Height);
413 }
414
415 // Thumb
416 UpdateThumb();
417
418 Invalidate();
419 }

421 private void UpdateThumb(bool forceRefresh = false)
422 {
423 if (ViewSize >= Maximum)
424 return;
425
426 // Cap to maximum value
427 var maximumValue = Maximum - ViewSize;
428 if (Value > maximumValue)
429 Value = maximumValue;
430
431 // Calculate size ratio
432 _viewContentRatio = (float)ViewSize / (float)Maximum;
433 var viewAreaSize = Maximum - ViewSize;
434 var positionRatio = (float)Value / (float)viewAreaSize;
435
436 // Update area
437 if (_scrollOrientation == VitNX_ScrollOrientation.Vertical)
438 {
439 var thumbSize = (int)(_trackArea.Height * _viewContentRatio);
440
441 if (thumbSize < Constsants.MinimumThumbSize)
442 thumbSize = Constsants.MinimumThumbSize;
443
444 var trackAreaSize = _trackArea.Height - thumbSize;
445 var thumbPosition = (int)(trackAreaSize * positionRatio);
446
447 _thumbArea = new Rectangle(_trackArea.Left + 3, _trackArea.Top + thumbPosition, Constsants.ScrollBarSize - 6, thumbSize);
448 }
449 else if (_scrollOrientation == VitNX_ScrollOrientation.Horizontal)
450 {
451 var thumbSize = (int)(_trackArea.Width * _viewContentRatio);
452
453 if (thumbSize < Constsants.MinimumThumbSize)
454 thumbSize = Constsants.MinimumThumbSize;
455
456 var trackAreaSize = _trackArea.Width - thumbSize;
457 var thumbPosition = (int)(trackAreaSize * positionRatio);
458
459 _thumbArea = new Rectangle(_trackArea.Left + thumbPosition, _trackArea.Top + 3, thumbSize, Constsants.ScrollBarSize - 6);
460 }
461
462 if (forceRefresh)
463 {
464 Invalidate();
465 Update();
466 }
467 }
