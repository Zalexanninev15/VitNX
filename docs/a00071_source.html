246 protected override void OnPaint(PaintEventArgs e)
247 {
248 var g = e.Graphics;
249 var rect = new Rectangle(0, 0, ClientSize.Width, ClientSize.Height);
250 var textColor = Colors.LightText;
251 var borderColor = Colors.GreySelection;
252 var fillColor = _isDefault ? Colors.VitNXBlueBackground : Colors.LightBackground;
253 if (Enabled)
254 {
255 if (ButtonStyle == VitNX_ButtonStyle.Normal)
256 {
257 if (Focused && TabStop) { borderColor = Colors.BlueHighlight; }
258 switch (ButtonState)
259 {
260 case VitNX_ControlState.Hover:
261 fillColor = _isDefault ? Colors.BlueBackground : Colors.LighterBackground;
262 break;
263
264 case VitNX_ControlState.Pressed:
265 fillColor = _isDefault ? Colors.VitNXBackground : Colors.VitNXBackground;
266 break;
267 }
268 }
269 else if (ButtonStyle == VitNX_ButtonStyle.Flat)
270 {
271 switch (ButtonState)
272 {
273 case VitNX_ControlState.Normal:
274 fillColor = Colors.GreyBackground;
275 break;
276
277 case VitNX_ControlState.Hover:
278 fillColor = Colors.MediumBackground;
279 break;
280
281 case VitNX_ControlState.Pressed:
282 fillColor = Colors.VitNXBackground;
283 break;
284 }
285 }
286 }
287 else
288 {
289 textColor = Colors.DisabledText;
290 fillColor = Colors.VitNXGreySelection;
291 }
292 using (var b = new SolidBrush(fillColor)) { g.FillRectangle(b, rect); }
293 if (ButtonStyle == VitNX_ButtonStyle.Normal)
294 {
295 using (var p = new Pen(borderColor, 1))
296 {
297 var modRect = new Rectangle(rect.Left, rect.Top, rect.Width - 1, rect.Height - 1);
298 g.DrawRectangle(p, modRect);
299 }
300 }
301 var textOffsetX = 0;
302 var textOffsetY = 0;
303 if (Image != null)
304 {
305 var stringSize = g.MeasureString(Text, Font, rect.Size);
306 var x = (ClientSize.Width / 2) - (Image.Size.Width / 2);
307 var y = (ClientSize.Height / 2) - (Image.Size.Height / 2);
308 switch (TextImageRelation)
309 {
310 case TextImageRelation.ImageAboveText:
311 textOffsetY = (Image.Size.Height / 2) + (ImagePadding / 2);
312 y = y - ((int)(stringSize.Height / 2) + (ImagePadding / 2));
313 break;
314
315 case TextImageRelation.TextAboveImage:
316 textOffsetY = ((Image.Size.Height / 2) + (ImagePadding / 2)) * -1;
317 y = y + ((int)(stringSize.Height / 2) + (ImagePadding / 2));
318 break;
319
320 case TextImageRelation.ImageBeforeText:
321 textOffsetX = Image.Size.Width + (ImagePadding * 2);
322 x = ImagePadding;
323 break;
324
325 case TextImageRelation.TextBeforeImage:
326 x = x + (int)stringSize.Width;
327 break;
328 }
329 g.DrawImageUnscaled(Image, x, y);
330 }
331 using (var b = new SolidBrush(textColor))
332 {
333 var modRect = new Rectangle(rect.Left + textOffsetX + Padding.Left, rect.Top + textOffsetY + Padding.Top, rect.Width - Padding.Horizontal, rect.Height - Padding.Vertical);
334 var stringFormat = new StringFormat
335 {
336 LineAlignment = StringAlignment.Center,
337 Alignment = StringAlignment.Center,
338 Trimming = StringTrimming.EllipsisCharacter
339 };
340 g.DrawString(Text, Font, b, modRect, stringFormat);
341 }
342 }
