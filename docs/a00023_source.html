140 private Bitmap GetImageInternal(Properties props)
141 {
142 var size = GetFontIconRealSize(props.Size, (int)props.Type);
143 var bmpTemp = new Bitmap(size.Width, size.Height);
144 using (Graphics g1 = Graphics.FromImage(bmpTemp))
145 {
146 g1.TextRenderingHint = TextRenderingHint.AntiAlias;
147 g1.Clear(Color.Transparent);
148 var font = GetIconFont(props.Size);
149 string character = char.ConvertFromUtf32((int)props.Type);
150 var format = new StringFormat()
151 {
152 Alignment = StringAlignment.Center,
153 LineAlignment = StringAlignment.Center,
154 Trimming = StringTrimming.Character
155 };
156 g1.DrawString(character, font, new SolidBrush(props.ForeColor), 0, 0);
157 g1.DrawImage(bmpTemp, 0, 0);
158 }
159 var bmp = ResizeImage(bmpTemp, props);
160 if (props.ShowBorder)
161 {
162 using (Graphics g2 = Graphics.FromImage(bmp))
163 {
164 var pen = new Pen(props.BorderColor, 1);
165 var borderRect = new Rectangle(0, 0, (int)(props.Size - pen.Width), (int)(props.Size - pen.Width));
166 g2.DrawRectangle(pen, borderRect);
167 }
168 }
169 return bmp;
170 }
171
172 private Size GetFontIconRealSize(int size, int iconIndex)
173 {
174 var bmpTemp = new Bitmap(size, size);
175 using (Graphics g1 = Graphics.FromImage(bmpTemp))
176 {
177 g1.TextRenderingHint = TextRenderingHint.AntiAlias;
178 g1.PixelOffsetMode = PixelOffsetMode.HighQuality;
179 var font = GetIconFont(size);
180 string character = char.ConvertFromUtf32(iconIndex);
181 var format = new StringFormat()
182 {
183 Alignment = StringAlignment.Center,
184 LineAlignment = StringAlignment.Center,
185 Trimming = StringTrimming.Word
186 };
187 var sizeF = g1.MeasureString(character, font, new Point(0, 0), format);
188 return sizeF.ToSize();
189 }
190 }

192 private Bitmap ResizeImage(Bitmap imgToResize, Properties props)
193 {
194 var srcWidth = imgToResize.Width;
195 var srcHeight = imgToResize.Height;
196 float ratio = (srcWidth > srcHeight) ? (srcWidth / (float)srcHeight) : (srcHeight / (float)srcWidth);
197 var dstWidth = (int)Math.Ceiling(srcWidth / ratio);
198 var dstHeight = (int)Math.Ceiling(srcHeight / ratio);
199 var x = (int)Math.Round((props.Size - dstWidth) / 2f, 0);
200 var y = (int)(1 + Math.Round((props.Size - dstHeight) / 2f, 0));
201 Bitmap b = new Bitmap(props.Size + props.Location.X, props.Size + props.Location.Y);
202 using (Graphics g = Graphics.FromImage(b))
203 {
204 g.Clear(props.BackColor);
205 g.SmoothingMode = SmoothingMode.HighQuality;
206 g.InterpolationMode = InterpolationMode.HighQualityBicubic;
207 g.PixelOffsetMode = PixelOffsetMode.HighQuality;
208 g.DrawImage(imgToResize, x + props.Location.X, y + props.Location.Y, dstWidth, dstHeight);
209 }
210 return b;
211 }

213 private void LoadFont()
214 {
215 try
216 {
217 if (!File.Exists($"{Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments)}\\{FONT_FILE_NAME}") && !string.IsNullOrEmpty(_downloadLink))
218 {
219 Uri downloadUri;
220 if (Uri.TryCreate(_downloadLink, UriKind.Absolute, out downloadUri)
221 && (downloadUri.Scheme == Uri.UriSchemeHttp || downloadUri.Scheme == Uri.UriSchemeHttps))
222 {
223 using (var client = new System.Net.WebClient())
224 client.DownloadFile(downloadUri, $"{Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments)}\\{FONT_FILE_NAME}");
225 }
226 }
227 }
228 finally
229 {
230 if (File.Exists($"{Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments)}\\{FONT_FILE_NAME}"))
231 _fonts.AddFontFile($"{Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments)}\\{FONT_FILE_NAME}");
232 }
233 }

235 private Font GetIconFont(int pixelSize)
236 {
237 var size = pixelSize / (16f / 12f);
238 var font = new Font(_fonts.Families[0], size, FontStyle.Regular, GraphicsUnit.Point);
239 return font;
240 }
