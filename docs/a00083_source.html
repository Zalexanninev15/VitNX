100 private void PaintCombobox()
101 {
102 if (_buffer == null) { _buffer = new Bitmap(ClientRectangle.Width, ClientRectangle.Height); }
103 using (var g = Graphics.FromImage(_buffer))
104 {
105 var rect = new Rectangle(0, 0, ClientSize.Width, ClientSize.Height);
106 var textColor = Colors.LightText;
107 var borderColor = Colors.GreySelection;
108 var fillColor = Colors.LightBackground;
109 if (Focused && TabStop) { borderColor = Colors.BlueHighlight; }
110 using (var b = new SolidBrush(fillColor)) { g.FillRectangle(b, rect); }
111 using (var p = new Pen(borderColor, 1)) { var modRect = new Rectangle(rect.Left, rect.Top, rect.Width - 1, rect.Height - 1); g.DrawRectangle(p, modRect); }
112 var icon = ScrollIcons.scrollbar_arrow_hot;
113 g.DrawImageUnscaled(icon, rect.Right - icon.Width - (Constsants.Padding / 2), (rect.Height / 2) - (icon.Height / 2));
114 var text = SelectedItem != null ? SelectedItem.ToString() : Text;
115 using (var b = new SolidBrush(textColor))
116 {
117 var padding = 2;
118 var modRect = new Rectangle(rect.Left + padding, rect.Top + padding, rect.Width - icon.Width - (Constsants.Padding / 2) - (padding * 2), rect.Height - (padding * 2));
119 var stringFormat = new StringFormat
120 {
121 LineAlignment = StringAlignment.Center,
122 Alignment = StringAlignment.Near,
123 FormatFlags = StringFormatFlags.NoWrap,
124 Trimming = StringTrimming.EllipsisCharacter
125 };
126 g.DrawString(text, Font, b, modRect, stringFormat);
127 }
128 }
129 }
