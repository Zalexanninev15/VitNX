136 private void UpdateScrollBars()
137 {
138 if (_vScrollBar.Maximum != ContentSize.Height)
139 _vScrollBar.Maximum = ContentSize.Height;
140
141 if (_hScrollBar.Maximum != ContentSize.Width)
142 _hScrollBar.Maximum = ContentSize.Width;
143
144 var scrollSize = Constsants.ScrollBarSize;
145
146 _vScrollBar.Location = new Point(ClientSize.Width - scrollSize, 0);
147 _vScrollBar.Size = new Size(scrollSize, ClientSize.Height);
148
149 _hScrollBar.Location = new Point(0, ClientSize.Height - scrollSize);
150 _hScrollBar.Size = new Size(ClientSize.Width, scrollSize);
151
152 if (DesignMode)
153 return;
154
155 // Do this twice in case changing the visibility of the scrollbars
156 // causes the VisibleSize to change in such a way as to require a second scrollbar.
157 // Probably a better way to detect that scenario...
158 SetVisibleSize();
159 SetScrollBarVisibility();
160 SetVisibleSize();
161 SetScrollBarVisibility();
162
163 if (_vScrollBar.Visible)
164 _hScrollBar.Width -= scrollSize;
165
166 if (_hScrollBar.Visible)
167 _vScrollBar.Height -= scrollSize;
168
169 _vScrollBar.ViewSize = _visibleSize.Height;
170 _hScrollBar.ViewSize = _visibleSize.Width;
171
172 UpdateViewport();
173 }
174
175 private void SetScrollBarVisibility()
176 {
177 _vScrollBar.Enabled = _visibleSize.Height < ContentSize.Height;
178 _hScrollBar.Enabled = _visibleSize.Width < ContentSize.Width;
179
180 if (_hideScrollBars)
181 {
182 _vScrollBar.Visible = _vScrollBar.Enabled;
183 _hScrollBar.Visible = _hScrollBar.Enabled;
184 }
185 }

195
196 if (_hScrollBar.Visible)
197 _visibleSize.Height -= scrollSize;
198 }

200 private void UpdateViewport()
201 {
202 var left = 0;
203 var top = 0;
204 var width = ClientSize.Width;
205 var height = ClientSize.Height;
206
207 if (_hScrollBar.Visible)
208 {
209 left = _hScrollBar.Value;
210 height -= _hScrollBar.Height;
211 }
212
213 if (_vScrollBar.Visible)
214 {
215 top = _vScrollBar.Value;
216 width -= _vScrollBar.Width;
217 }
218
219 Viewport = new Rectangle(left, top, width, height);
220
221 var pos = PointToClient(MousePosition);
222 _offsetMousePosition = new Point(pos.X + Viewport.Left, pos.Y + Viewport.Top);
223
224 Invalidate();
225 }

314 protected override void OnMouseWheel(MouseEventArgs e)
315 {
316 base.OnMouseWheel(e);
317
318 var horizontal = false;
319
320 if (_hScrollBar.Visible && ModifierKeys == Keys.Control)
321 horizontal = true;
322
323 if (_hScrollBar.Visible && !_vScrollBar.Visible)
324 horizontal = true;
325
326 if (!horizontal)
327 {
328 if (e.Delta > 0)
329 _vScrollBar.ScrollByPhysical(3);
330 else if (e.Delta < 0)
331 _vScrollBar.ScrollByPhysical(-3);
332 }
333 else
334 {
335 if (e.Delta > 0)
336 _hScrollBar.ScrollByPhysical(3);
337 else if (e.Delta < 0)
338 _hScrollBar.ScrollByPhysical(-3);
339 }

358 private void DragTimer_Tick(object sender, EventArgs e)
359 {
360 var pos = PointToClient(MousePosition);
361
362 var right = ClientRectangle.Right;
363 var bottom = ClientRectangle.Bottom;
364
365 if (_vScrollBar.Visible)
366 right = _vScrollBar.Left;
367
368 if (_hScrollBar.Visible)
369 bottom = _hScrollBar.Top;
370
371 if (_vScrollBar.Visible)
372 {
373 // Scroll up
374 if (pos.Y < ClientRectangle.Top)
375 {
376 var difference = (pos.Y - ClientRectangle.Top) * -1;
377
378 if (MaxDragChange > 0 && difference > MaxDragChange)
379 difference = MaxDragChange;
380
381 _vScrollBar.Value = _vScrollBar.Value - difference;
382 }
383
384 // Scroll down
385 if (pos.Y > bottom)
386 {
387 var difference = pos.Y - bottom;
388
389 if (MaxDragChange > 0 && difference > MaxDragChange)
390 difference = MaxDragChange;
391
392 _vScrollBar.Value = _vScrollBar.Value + difference;
393 }
394 }
395
396 if (_hScrollBar.Visible)
397 {
398 // Scroll left
399 if (pos.X < ClientRectangle.Left)
400 {
401 var difference = (pos.X - ClientRectangle.Left) * -1;
402
403 if (MaxDragChange > 0 && difference > MaxDragChange)
404 difference = MaxDragChange;
405
406 _hScrollBar.Value = _hScrollBar.Value - difference;
407 }
408
409 // Scroll right
410 if (pos.X > right)
411 {
412 var difference = pos.X - right;
413
414 if (MaxDragChange > 0 && difference > MaxDragChange)
415 difference = MaxDragChange;
416
417 _hScrollBar.Value = _hScrollBar.Value + difference;
418 }
419 }
