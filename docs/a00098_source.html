146 private ToolStripMenuItem GetMenuItem(VitNX_DropdownItem item)
147 {
148 foreach (ToolStripMenuItem menuItem in _menu.Items)
149 {
150 if ((VitNX_DropdownItem)menuItem.Tag == item)
151 return menuItem;
152 }
153
154 return null;
155 }

169 private void ShowMenu()
170 {
171 if (_menu.Visible)
172 return;
173
174 SetControlState(VitNX_ControlState.Pressed);
175
176 _menuOpen = true;
177
178 var pos = new Point(0, ClientRectangle.Bottom);
179
180 if (_dropdownDirection == ToolStripDropDownDirection.AboveLeft || _dropdownDirection == ToolStripDropDownDirection.AboveRight)
181 pos.Y = 0;
182
183 _menu.Show(this, pos, _dropdownDirection);
184
185 if (SelectedItem != null)
186 {
187 var selectedItem = GetMenuItem(SelectedItem);
188 selectedItem.Select();
189 }
190 }
191
192 private void ResizeMenu()
193 {
194 var width = ClientRectangle.Width;
195 var height = (_menu.Items.Count * _itemHeight) + 4;
196
197 if (height > _maxHeight)
198 height = _maxHeight;
199
200 // Dirty: Check what the autosized items are
201 foreach (ToolStripMenuItem item in _menu.Items)
202 {
203 item.AutoSize = true;
204
205 if (item.Size.Width > width)
206 width = item.Size.Width;
207
208 item.AutoSize = false;
209 }
210
211 // Force the size
212 foreach (ToolStripMenuItem item in _menu.Items)
213 item.Size = new Size(width - 1, _itemHeight);
214
215 _menu.Size = new Size(width, height);
216 }

222 private void Items_CollectionChanged(object sender, NotifyCollectionChangedEventArgs e)
223 {
224 if (e.Action == NotifyCollectionChangedAction.Add)
225 {
226 foreach (VitNX_DropdownItem item in e.NewItems)
227 {
228 var menuItem = new ToolStripMenuItem(item.Text)
229 {
230 Image = item.Icon,
231 AutoSize = false,
232 Height = _itemHeight,
233 Font = Font,
234 Tag = item,
235 TextAlign = ContentAlignment.MiddleLeft
236 };
237
238 _menu.Items.Add(menuItem);
239 menuItem.Click += Item_Select;
240
241 if (SelectedItem == null)
242 SelectedItem = item;
243 }
244 }
245
246 if (e.Action == NotifyCollectionChangedAction.Remove)
247 {
248 foreach (VitNX_DropdownItem item in e.OldItems)
249 {
250 foreach (ToolStripMenuItem menuItem in _menu.Items)
251 {
252 if ((VitNX_DropdownItem)menuItem.Tag == item)
253 _menu.Items.Remove(menuItem);
254 }
255 }
256 }
257
258 if (e.Action == NotifyCollectionChangedAction.Reset)
259 {
260 _menu.Items.Clear();
261 SelectedItem = null;
262 }
263
264 ResizeMenu();
265 }

267 private void Item_Select(object sender, EventArgs e)
268 {
269 var menuItem = sender as ToolStripMenuItem;

274 if (_selectedItem != dropdownItem)
275 SelectedItem = dropdownItem;

278 private void VitNX_DropdownList_SelectedItemChanged(object sender, EventArgs e)
279 {
280 foreach (ToolStripMenuItem item in _menu.Items)
281 {
282 if ((VitNX_DropdownItem)item.Tag == SelectedItem)
283 {
284 item.BackColor = Colors.VitNXBlueBackground;
285 item.Font = new Font(Font, FontStyle.Bold);
286 }
287 else
288 {
289 item.BackColor = Colors.GreyBackground;
290 item.Font = new Font(Font, FontStyle.Regular);
291 }
292 }
293
294 Invalidate();

393 protected override void OnPaint(PaintEventArgs e)
394 {
395 var g = e.Graphics;
396
397 // Draw background
398 using (var b = new SolidBrush(Colors.MediumBackground))
399 {
400 g.FillRectangle(b, ClientRectangle);
401 }
402
403 // Draw normal state
404 if (ControlState == VitNX_ControlState.Normal)
405 {
406 if (ShowBorder)
407 {
408 using (var p = new Pen(Colors.LightBorder, 1))
409 {
410 var modRect = new Rectangle(ClientRectangle.Left, ClientRectangle.Top, ClientRectangle.Width - 1, ClientRectangle.Height - 1);
411 g.DrawRectangle(p, modRect);
412 }
413 }
414 }
415
416 // Draw hover state
417 if (ControlState == VitNX_ControlState.Hover)
418 {
419 using (var b = new SolidBrush(Colors.VitNXBorder))
420 {
421 g.FillRectangle(b, ClientRectangle);
422 }
423
424 using (var b = new SolidBrush(Colors.VitNXBackground))
425 {
426 var arrowRect = new Rectangle(ClientRectangle.Right - DropdownIcons.small_arrow.Width - 8, ClientRectangle.Top, DropdownIcons.small_arrow.Width + 8, ClientRectangle.Height);
427 g.FillRectangle(b, arrowRect);
428 }
429
430 using (var p = new Pen(Colors.BlueSelection, 1))
431 {
432 var modRect = new Rectangle(ClientRectangle.Left, ClientRectangle.Top, ClientRectangle.Width - 1 - DropdownIcons.small_arrow.Width - 8, ClientRectangle.Height - 1);
433 g.DrawRectangle(p, modRect);
434 }
435 }
436
437 // Draw pressed state
438 if (ControlState == VitNX_ControlState.Pressed)
439 {
440 using (var b = new SolidBrush(Colors.VitNXBorder))
441 {
442 g.FillRectangle(b, ClientRectangle);
443 }
444
445 using (var b = new SolidBrush(Colors.BlueSelection))
446 {
447 var arrowRect = new Rectangle(ClientRectangle.Right - DropdownIcons.small_arrow.Width - 8, ClientRectangle.Top, DropdownIcons.small_arrow.Width + 8, ClientRectangle.Height);
448 g.FillRectangle(b, arrowRect);
449 }
450 }
451
452 // Draw dropdown arrow
453 using (var img = DropdownIcons.small_arrow)
454 {
455 g.DrawImageUnscaled(img, ClientRectangle.Right - img.Width - 4, ClientRectangle.Top + (ClientRectangle.Height / 2) - (img.Height / 2));
456 }
457
458 // Draw selected item
459 if (SelectedItem != null)
460 {
461 // Draw Icon
462 var hasIcon = SelectedItem.Icon != null;
463
464 if (hasIcon)
465 {
466 g.DrawImageUnscaled(SelectedItem.Icon, new Point(ClientRectangle.Left + 5, ClientRectangle.Top + (ClientRectangle.Height / 2) - (_iconSize / 2)));
467 }
468
469 // Draw Text
470 using (var b = new SolidBrush(Colors.LightText))
471 {
472 var stringFormat = new StringFormat
473 {
474 Alignment = StringAlignment.Near,
475 LineAlignment = StringAlignment.Center
476 };
477
478 var rect = new Rectangle(ClientRectangle.Left + 2, ClientRectangle.Top, ClientRectangle.Width - 16, ClientRectangle.Height);
479
480 if (hasIcon)
481 {
482 rect.X += _iconSize + 7;
483 rect.Width -= _iconSize + 7;
484 }
485
486 g.DrawString(SelectedItem.Text, Font, b, rect, stringFormat);
487 }
488 }
