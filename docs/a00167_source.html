66 Height = DockPanel.Height

68

71
72 break;
73
74 case VitNX_DockArea.Right:
75

78 X = DockPanel.PointToScreen(Point.Empty).X + DockPanel.Width - 50,
79 Y = DockPanel.PointToScreen(Point.Empty).Y,
80 Width = 50,

82 };

84 DropArea = rightRect;

87 break;

93

97 width -= DockPanel.Regions[VitNX_DockArea.Left].Width;

99

101 {
102 width -= DockPanel.Regions[VitNX_DockArea.Right].Width;
103 }

120 private void BuildGroupAreas()

125 var dropRect = new Rectangle

128 Y = DockGroup.PointToScreen(Point.Empty).Y,
129 Width = DockGroup.Width,
130 Height = DockGroup.Height
131 };
132
133 DropArea = dropRect;
134 HighlightArea = dropRect;
135
136 break;
137
138 case DockInsertType.Before:
139 var beforeDropWidth = DockGroup.Width;
140 var beforeDropHeight = DockGroup.Height;
141
142 switch (DockGroup.DockArea)
143 {
144 case VitNX_DockArea.Left:
145 case VitNX_DockArea.Right:
146 beforeDropHeight = DockGroup.Height / 4;
147 break;
148
149 case VitNX_DockArea.Bottom:
150 beforeDropWidth = DockGroup.Width / 4;
151 break;
152 }
153
154 var beforeDropRect = new Rectangle
155 {
156 X = DockGroup.PointToScreen(Point.Empty).X,
157 Y = DockGroup.PointToScreen(Point.Empty).Y,
158 Width = beforeDropWidth,
159 Height = beforeDropHeight
160 };
161
162 DropArea = beforeDropRect;
163 HighlightArea = beforeDropRect;
164
165 break;
166
167 case DockInsertType.After:
168 var afterDropX = DockGroup.PointToScreen(Point.Empty).X;
169 var afterDropY = DockGroup.PointToScreen(Point.Empty).Y;
170 var afterDropWidth = DockGroup.Width;
171 var afterDropHeight = DockGroup.Height;
172
173 switch (DockGroup.DockArea)
174 {
175 case VitNX_DockArea.Left:
176 case VitNX_DockArea.Right:
177 afterDropHeight = DockGroup.Height / 4;
178 afterDropY = DockGroup.PointToScreen(Point.Empty).Y + DockGroup.Height - afterDropHeight;
179 break;
180
181 case VitNX_DockArea.Bottom:
182 afterDropWidth = DockGroup.Width / 4;
183 afterDropX = DockGroup.PointToScreen(Point.Empty).X + DockGroup.Width - afterDropWidth;
184 break;
185 }
186
187 var afterDropRect = new Rectangle
188 {
189 X = afterDropX,
190 Y = afterDropY,
191 Width = afterDropWidth,
192 Height = afterDropHeight
193 };
194
