14 {
15 string[] folders = Directory.GetDirectories(sourceFolder);

17 string[] files = Directory.GetFiles(sourceFolder);

20 catch (Exception e) { ls.Add(e.Message.Trim('\n')); }

79 using (var md5 = System.Security.Cryptography.MD5.Create())
80 {
81 using (var Stream = File.OpenRead(targetFile))
82 {
83 var hashBytes = md5.ComputeHash(Stream);

110 {
111 string bak_file = sourceFolder + "." + newFileExtension.Replace(".", "");
112 if (File.Exists(bak_file)) { try { File.Delete(bak_file); } catch { } }
113 if (saveOldFile == false) { try { File.Move(sourceFolder, bak_file); } catch { } }
114 else
115 {
116 File.Copy(sourceFolder, bak_file);
117 try { File.Delete(sourceFolder); } catch { }
