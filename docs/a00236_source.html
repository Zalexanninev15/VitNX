136 private void CalculateSize()
137 {
138 var width = 260; var height = 124;
139
140 // Reset form back to original size
141 Size = new Size(width, height);
142
143 lblText.Text = string.Empty;
144 lblText.AutoSize = true;
145 lblText.Text = _message;
146
147 // Set the minimum dialog size to whichever is bigger - the original size or the buttons.
148 var minWidth = Math.Max(width, TotalButtonSize + 15);
149
150 // Calculate the total size of the message
151 var totalWidth = lblText.Right + 25;
152
153 // Make sure we're not making the dialog bigger than the maximum size
154 if (totalWidth < _maximumWidth)
155 {
156 // Width is smaller than the maximum width.
157 // This means we can have a single-line message box.
158 // Move the label to accomodate this.
159 width = totalWidth;
160 lblText.Top = picIcon.Top + (picIcon.Height / 2) - (lblText.Height / 2);
161 }
162 else
163 {
164 // Width is larger than the maximum width.
165 // Change the label size and wrap it.
166 width = _maximumWidth;
167 var offsetHeight = Height - picIcon.Height;
168 lblText.AutoUpdateHeight = true;
169 lblText.Width = width - lblText.Left - 25;
170 height = offsetHeight + lblText.Height;
171 }
172
173 // Force the width to the minimum width
174 if (width < minWidth)
175 width = minWidth;
176
177 // Set the new size of the dialog
178 Size = new Size(width, height);
179 }
