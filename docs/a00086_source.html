279 protected override void OnPaint(PaintEventArgs e)
280 {

282 var rect = new Rectangle(0, 0, ClientSize.Width, ClientSize.Height);
283
284 var size = Constsants.CheckBoxSize;
285
286 var textColor = Colors.LightText;

288 var fillColor = Colors.LightestBackground;
289
290 if (Enabled)
291 {

294 borderColor = Colors.BlueHighlight;
295 fillColor = Colors.BlueSelection;
296 }
297
298 if (_controlState == VitNX_ControlState.Hover)
299 {
300 borderColor = Colors.BlueHighlight;
301 fillColor = Colors.BlueSelection;
302 }
303 else if (_controlState == VitNX_ControlState.Pressed)
304 {
305 borderColor = Colors.GreyHighlight;
306 fillColor = Colors.GreySelection;
307 }
308 }
309 else
310 {
311 textColor = Colors.DisabledText;
312 borderColor = Colors.GreyHighlight;
313 fillColor = Colors.GreySelection;
314 }
315
316 using (var b = new SolidBrush(Colors.GreyBackground))
317 {
318 g.FillRectangle(b, rect);
319 }
320
321 using (var p = new Pen(borderColor))
322 {
323 var boxRect = new Rectangle(0, (rect.Height / 2) - (size / 2), size, size);
324 g.DrawRectangle(p, boxRect);
325 }
326
327 if (Checked)
328 {
329 using (var b = new SolidBrush(fillColor))
330 {
331 Rectangle boxRect = new Rectangle(2, (rect.Height / 2) - ((size - 4) / 2), size - 3, size - 3);
332 g.FillRectangle(b, boxRect);
333 }
334 }
335
336 using (var b = new SolidBrush(textColor))
337 {
338 var stringFormat = new StringFormat
339 {
340 LineAlignment = StringAlignment.Center,
341 Alignment = StringAlignment.Near
342 };
343
344 var modRect = new Rectangle(size + 4, 0, rect.Width - size, rect.Height);
345 g.DrawString(Text, Font, b, modRect, stringFormat);
346 }
