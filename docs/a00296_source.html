195 protected override void OnPaint(PaintEventArgs e)
196 {
197 base.OnPaint(e);
198 Graphics graph = e.Graphics;
199 if (borderRadius > 1)
200 {
201 var rectBorderSmooth = ClientRectangle;
202 var rectBorder = Rectangle.Inflate(rectBorderSmooth, -borderSize, -borderSize);
203 int smoothSize = borderSize > 0 ? borderSize : 1;
204 using (GraphicsPath pathBorderSmooth = GetFigurePath(rectBorderSmooth, borderRadius))
205 using (GraphicsPath pathBorder = GetFigurePath(rectBorder, borderRadius - borderSize))
206 using (Pen penBorderSmooth = new Pen(Parent.BackColor, smoothSize))
207 using (Pen penBorder = new Pen(borderColor, borderSize))
208 {
209 Region = new Region(pathBorderSmooth);
210 if (borderRadius > 15)
211 SetTextBoxRoundedRegion();
212 graph.SmoothingMode = SmoothingMode.AntiAlias;
213 penBorder.Alignment = PenAlignment.Center;
214 if (isFocused)
215 penBorder.Color = borderFocusColor;
216 if (underlinedStyle)
217 {
218 graph.DrawPath(penBorderSmooth, pathBorderSmooth);
219 graph.SmoothingMode = SmoothingMode.None;
220 graph.DrawLine(penBorder, 0, Height - 1, Width, Height - 1);
221 }
222 else
223 {
224 graph.DrawPath(penBorderSmooth, pathBorderSmooth);
225 graph.DrawPath(penBorder, pathBorder);
226 }
227 }
228 }
229 else
230 {
231 using (Pen penBorder = new Pen(borderColor, borderSize))
232 {
233 Region = new Region(ClientRectangle);
234 penBorder.Alignment = PenAlignment.Inset;
235 if (isFocused)
236 penBorder.Color = borderFocusColor;
237 if (underlinedStyle)
238 graph.DrawLine(penBorder, 0, Height - 1, Width, Height - 1);
239 else
240 graph.DrawRectangle(penBorder, 0, 0, Width - 0.5F, Height - 0.5F);
241 }
242 }

269 private GraphicsPath GetFigurePath(Rectangle rect, int radius)
270 {
271 GraphicsPath path = new GraphicsPath();
272 float curveSize = radius * 2F;
273 path.StartFigure();
274 path.AddArc(rect.X, rect.Y, curveSize, curveSize, 180, 90);
275 path.AddArc(rect.Right - curveSize, rect.Y, curveSize, curveSize, 270, 90);
276 path.AddArc(rect.Right - curveSize, rect.Bottom - curveSize, curveSize, curveSize, 0, 90);
277 path.AddArc(rect.X, rect.Bottom - curveSize, curveSize, curveSize, 90, 90);
278 path.CloseFigure();
279 return path;
280 }

283 {
284 GraphicsPath pathTxt;
