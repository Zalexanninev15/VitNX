110 public static string Post(string url, string options)
111 {
112 // Example: Post("https://site.com/auth", "client_id=43435&key=create");
113 try
114 {
115 WebRequest req = WebRequest.Create(url);
116 req.Method = "POST";
117 req.Timeout = 100000;
118 req.ContentType = "application/x-www-form-urlencoded";
119 byte[] sentData = Encoding.UTF8.GetBytes(options);
120 req.ContentLength = sentData.Length;
121 Stream sendStream = req.GetRequestStream();
122 sendStream.Write(sentData, 0, sentData.Length);
123 sendStream.Close();
124 WebResponse res = req.GetResponse();
125 Stream ReceiveStream = res.GetResponseStream();
126 StreamReader sr = new StreamReader(ReceiveStream, Encoding.UTF8);
127 char[] read = new char[256];
128 int count = sr.Read(read, 0, 256);
129 string Out = string.Empty;
130 while (count > 0)
131 {
132 string str = new string(read, 0, count);
133 Out += str;
134 count = sr.Read(read, 0, 256);
135 }
136 return Out;
137 }
138 catch (Exception ex) { return ex.Message; }
139 }
140 }
