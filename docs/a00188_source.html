208 private void CreateRegions()
209 {
210 var documentRegion = new VitNX_DockRegion(this, VitNX_DockArea.Document);
211 _regions.Add(VitNX_DockArea.Document, documentRegion);
212
213 var leftRegion = new VitNX_DockRegion(this, VitNX_DockArea.Left);
214 _regions.Add(VitNX_DockArea.Left, leftRegion);
215
216 var rightRegion = new VitNX_DockRegion(this, VitNX_DockArea.Right);
217 _regions.Add(VitNX_DockArea.Right, rightRegion);
218
219 var bottomRegion = new VitNX_DockRegion(this, VitNX_DockArea.Bottom);
220 _regions.Add(VitNX_DockArea.Bottom, bottomRegion);
221
222 // Add the regions in this order to force the bottom region to be positioned
223 // between the left and right regions properly.
224 Controls.Add(documentRegion);
225 Controls.Add(bottomRegion);
226 Controls.Add(leftRegion);
227 Controls.Add(rightRegion);
228
229 // Create tab index for intuitive tabbing order
230 documentRegion.TabIndex = 0;
231 rightRegion.TabIndex = 1;
232 bottomRegion.TabIndex = 2;
233 leftRegion.TabIndex = 3;
234 }

245 public DockPanelState GetDockPanelState()
246 {
247 var state = new DockPanelState();
248
249 state.Regions.Add(new DockRegionState(VitNX_DockArea.Document));
250 state.Regions.Add(new DockRegionState(VitNX_DockArea.Left, _regions[VitNX_DockArea.Left].Size));
251 state.Regions.Add(new DockRegionState(VitNX_DockArea.Right, _regions[VitNX_DockArea.Right].Size));
252 state.Regions.Add(new DockRegionState(VitNX_DockArea.Bottom, _regions[VitNX_DockArea.Bottom].Size));
253
254 var _groupStates = new Dictionary<VitNX_DockGroup, DockGroupState>();
255
256 var orderedContent = _contents.OrderBy(c => c.Order);
257 foreach (var content in orderedContent)
258 {
259 foreach (var region in state.Regions)
260 {
261 if (region.Area == content.DockArea)
262 {
263 DockGroupState groupState;
264
265 if (_groupStates.ContainsKey(content.DockGroup))
266 {
267 groupState = _groupStates[content.DockGroup];
268 }
269 else
270 {
271 groupState = new DockGroupState();
272 region.Groups.Add(groupState);
273 _groupStates.Add(content.DockGroup, groupState);
274 }
275
276 groupState.Contents.Add(content.SerializationKey);
277
278 groupState.VisibleContent = content.DockGroup.VisibleContent.SerializationKey;
279 }
280 }
281 }
282
283 return state;
284 }

286 public void RestoreDockPanelState(DockPanelState state, Func<string, VitNX_DockContent> getContentBySerializationKey)
287 {
288 foreach (var region in state.Regions)
289 {
290 switch (region.Area)
291 {
292 case VitNX_DockArea.Left:
293 _regions[VitNX_DockArea.Left].Size = region.Size;
294 break;
295
296 case VitNX_DockArea.Right:
297 _regions[VitNX_DockArea.Right].Size = region.Size;
298 break;
299
300 case VitNX_DockArea.Bottom:
301 _regions[VitNX_DockArea.Bottom].Size = region.Size;
302 break;
303 }
304
305 foreach (var group in region.Groups)
306 {
307 VitNX_DockContent previousContent = null;
308 VitNX_DockContent visibleContent = null;
309
310 foreach (var contentKey in group.Contents)
311 {
312 var content = getContentBySerializationKey(contentKey);
313
314 if (content == null)
315 continue;
316
317 content.DockArea = region.Area;
318
319 if (previousContent == null)
320 AddContent(content);
321 else
322 AddContent(content, previousContent.DockGroup);
323
324 previousContent = content;
325
326 if (group.VisibleContent == contentKey)
327 visibleContent = content;
328 }
329
330 if (visibleContent != null)
331 visibleContent.Select();
332 }
333 }
334 }
