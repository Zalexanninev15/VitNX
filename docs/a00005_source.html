10 public static List<string> GetItemsListInFolder(string sourceFolder)
11 {
12 List<string> ls = new List<string>();
13 try
14 {
15 string[] folders = Directory.GetDirectories(sourceFolder);
16 foreach (string folder in folders)
17 ls.Add("Folder: " + folder);
18 string[] files = Directory.GetFiles(sourceFolder);
19 foreach (string filename in files)
20 ls.Add("File: " + filename);
21 }
22 catch (Exception e) { ls.Add(e.Message.Trim('\n')); }
23 return ls;
24 }

80 {
81 using (var md5 = System.Security.Cryptography.MD5.Create())
82 {
83 using (var Stream = File.OpenRead(targetFile))
84 {
85 var hashBytes = md5.ComputeHash(Stream);
86 var sb = new StringBuilder();
87 foreach (var t in hashBytes)
88 sb.Append(t.ToString("X2"));
89 return Convert.ToString(sb);

112 {
113 string bak_file = sourceFolder + "." + newFileExtension.Replace(".", "");
114 if (File.Exists(bak_file)) { try { File.Delete(bak_file); } catch { } }
115 if (saveOldFile == false) { try { File.Move(sourceFolder, bak_file); } catch { } }
116 else
117 {
118 File.Copy(sourceFolder, bak_file);
119 try { File.Delete(sourceFolder); } catch { }
